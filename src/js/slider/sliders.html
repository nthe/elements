Create element

        const slider = Slider('slider-1', document.body);
        

        Customize (optional)
        
        slider.configure({
            value: 0.5,                 // initial value
            width: 120,                 // width of slider
            lineWidth: 10,              // width of line
            vertical: false,            // verical orientation
            handleRadius: 0,            // radius of handle
            showHandle: false,          // toggle handle
            handleColor: '#fff',        // color of handle
            rangeColor: '#0af',         // color of range indicator
            indicatorColor: '#fc1',     // color of value indicator
            backdropColor: '#424242'    // color of backdrop
        });

        

        Add listener (optional)
        
        slider.onChange = (msg) => console.log(msg);
