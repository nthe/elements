Create element

        const knob = Knob('knob-1', document.body);
        

        Customize (optional)
        
        knob.configure({
            value: 0.5,                 // initial value
            radius: 120,                // radius of knob
            lineWidth: 10,              // width of line
            startAngle: 180,            // start angle
            sweepAngle: 180,            // sweep angle
            fontSize: 24,               // font size
            fontFamily: 'segoe ui',     // font family
            rangeColor: '#0af',         // color of range indicator
            indicatorColor: '#fc1',     // color of value indicator
            backdropColor: '#424242'    // color of backdrop
        });
        

        Add listener (optional)
        
        knob.onChange = (msg) => console.log(msg);
